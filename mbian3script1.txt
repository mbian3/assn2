create table agent (agentID varchar(2), FName varchar(30), LName varchar(30), city varchar(15), comission int, primary key (agentID));
create table customer (customerID varchar(2), FName varchar(30), LName varchar(30), city varchar(15), phone varchar(15), aID varchar(2), primary key(customerID), foreign key (aID) REFERENCES agent(agentID)); 
create table product (productID varchar(2), description varchar(20), cost float, quantity int, primary key (productID));
create table purchase (custID varchar(2), prodID varchar(20), quantityPurchased int, primary key (custID, prodID), foreign key (custID) REFERENCES customer (customerID), foreign key (prodID) REFERENCES product (productID));
